\quad dz = 0\\

\qquad z0 = Z.copy()\\

\qquad Z += A\_v * dt\\
\qquad Z = diffusion(e)\\

\qquad dz += Z - z0\\

\quad current\_time = t\\
\quad \textcolor{magentaDef}{return} dz
